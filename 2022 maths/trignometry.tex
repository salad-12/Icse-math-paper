 \begin{enumerate}
	 \item Prove that:\\
		 $\frac{1}{1 + \sin\theta}$ $+$ $\frac{1}{1 - \sin\theta}$ $=$ $2\sec^{2}\theta$
	 \item Prove that:\\
		 $\frac{(1 + \sin\theta)^{2} + (1 - \sin\theta)^{2}}{2\cos^{2}\theta}$ $=$ $\sec^{2}\theta + \tan^{2}\theta $
	 \item Prove that:\\
		 $1 + \frac{\tan^{2}\theta}{1 + \sec\theta} = \sec\theta$
 \end{enumerate}
